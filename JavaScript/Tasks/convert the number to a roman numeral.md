Задача на преоброзование наших чисел в римские.

#### Решение которое зашло мне:

```javaScript

function solution (number) {

  const  roman = {M:1000,CM:900, D:500,CD:400,C:100,XC:90,L:50,XL:40,X:10,IX:9,V:5,IV:4,I:1 }; // объект в котором хранятся наши ключи.
  
  let answer = ''; // здесь будет наш результат.
  
  while(number > 0) { // первый и основной цикл, который будет проверять, осталось ли еще что-то в передаваемом number
    for(a in roman) { // второй цикл, который перебирает объект roman
      if(roman[a] <= number){answer += a; number -= roman[a]; break;} // если значение ключа меньше нашего числа, то мы в answer добавляем ключ и из number вычитаем значение этого ключа.
              // например мы передаем 1520 в number. в roman[a] будет находиться 1000 под a будет M - значение и имя ключа. мы M влепим в answer, а из 1520 (наш number) вычтем 1000.
         // потом останавливаем цикл for, цикл while продолжит работу если в number еще что-то осталось и мы опять запустим цикл for.
    }
  }
  return answer;
}

```
